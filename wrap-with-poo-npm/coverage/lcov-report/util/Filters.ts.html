import Util from '@/util/Util';
 
class Filters {
  install(Vue) {
 
    Vue.filter('capitalize', function (value: string) {
      return Util.stringConvert.capitalize(value);
    });
 
    Vue.filter('dateTime', function (value: string | number) {
      if (value) {
        return Util.formatDate(value);
      } else {
        return '-';
      }
    });
 
    Vue.filter('workspaceNotActivated', function (value: string | number) {
      return value === '-' ? 'Not Yet Activated' : value;
    });
 
    Vue.filter('fileSize', function (value: number) {
      return Util.getFileSize(value);
    });
    Vue.filter('truncate', function (value: string, length = 10) {
      return Util.stringConvert.truncate(value, length);
    });
  }
}
 
export default new Filters();
