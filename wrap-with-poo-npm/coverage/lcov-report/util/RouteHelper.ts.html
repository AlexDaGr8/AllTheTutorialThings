import Core from '@/types/core.d.ts';
 
export default class RouteHelper {
  static getPageHeaderInfo(route, pageName, includePageTitle): Core.PageHeader {
    const defaultChildRoute: boolean = route.matched[0].path === route.path
      || `${route.matched[0].path}/` === route.path;
    const meta = defaultChildRoute
      ? route.matched[0].meta
      : route.meta;
    const title: string = includePageTitle
      ? `${meta.page_title} ${pageName}`
      : pageName;
    return {
      parent_path: defaultChildRoute ? '' : route.matched[0].path,
      parent_title: defaultChildRoute
        ? ''
        : route.matched[0].meta.page_title,
      title: title,
      description: meta.page_description,
      icon: meta.page_icon,
      image: meta.page_image,
      svg: meta.page_svg
    };
  }
}
