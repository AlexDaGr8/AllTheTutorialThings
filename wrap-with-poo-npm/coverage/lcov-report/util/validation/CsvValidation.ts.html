/* Vee-validate rules ref: http://vee-validate.logaretm.com/v2/guide/rules.html#length*/
export const role = {
  getMessage(validRoles: string[]) {
    const roles = validRoles.join(', ').toLowerCase();
    return `Invalid role value. Valid values are: ${roles}`;
  },
  validate(value: string, validRoles: string[]) {
    const okRole = validRoles.find(item => item.toLowerCase() === value);
    return !!okRole;
  }
};
 
export const username = {
  getMessage(value, args) {
    const usernameExists = args.find(item => item === value);
    value = value.length > 20 ? value.slice(0, 20) + '...' : value;
    if (usernameExists) {
      return `The user name, ${value}, is already being used.`;
    }
    return `Username ${value} is invalid.
      Usernames must be between 4-20 characters and consist of
      lowercase letters, numbers, dashes, underscores, and/or periods.
      Usernames may not begin with a dash or underscore, and may not end with a period.`;
  },
  validate(value, args) {
    const okUserName = /^[a-z0-9.]{1}[a-z0-9-_.]{2,18}[a-z0-9-_]{1}$/.test(
      value
    );
    const usernameExists = args.find(item => item === value);
    return okUserName && !usernameExists;
  }
};
 
export const email = {
  getMessage() {
    return 'Invalid email.';
  },
  validate(value) {
    const okEmail = /^(([^<>()\[\]\\.,;:\s@"]+(\.[^<>()\[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/.test(value);
    return okEmail;
  }
};
 
export const CSV_VALIDATION = {
  USERNAME: username,
  ROLE: role,
  EMAIL: email,
};
