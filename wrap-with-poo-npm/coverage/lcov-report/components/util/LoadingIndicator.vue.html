<template>
  <div class="loading-wrapper">
    <slot></slot>
 
    <div v-if="isLoading" class="loading-overlay">
      <div class="ui large loader"></div>
    </div>
  </div>
</template>
 
<script lang="ts">
import Vue from 'vue';
 
export default Vue.extend({
  props: ['loading'],
 
  watch: {
    loading(newVal, oldVal) {
      this.isLoading = newVal;
    }
  },
 
  data() {
    return {
      isLoading: this.loading
    };
  }
});
</script>
