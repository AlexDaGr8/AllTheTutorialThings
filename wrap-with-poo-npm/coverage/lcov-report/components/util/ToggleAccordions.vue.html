<template>
  <a :id="toggleId" href="javascript:;" @click="toggleAll(toggle)">
    <i class="icon" :class="toggle === 'close' ? 'minus' : 'plus'"></i>
    <span>{{ toggle === 'close' ? 'Collapse' : 'Expand' }} All</span>
  </a>
</template>
<script lang="ts">
import Vue from 'vue';
import $ from 'jquery';
 
export default Vue.extend({
  name: 'ToggleAccordions',
  props: {
    toggleId: {
      type: String
    }
  },
  data() {
    return {
      toggle: 'close'
    };
  },
  methods: {
    toggleAll(state) {
      const accordions = document.getElementsByClassName('accordion');
      Object.keys(accordions).forEach(key => {
        const title = accordions[key].getElementsByClassName('title')[0];
        const active = title.classList.contains('active');
        if (state === 'open' && !active) {
          title.click();
        } else if (state === 'close' && active) {
          title.click();
        }
      });
 
      this.toggle = state === 'open' ? 'close' : 'open';
    }
  }
});
</script>
 
<style scoped>
a {
  display: inline-block;
}
</style>
