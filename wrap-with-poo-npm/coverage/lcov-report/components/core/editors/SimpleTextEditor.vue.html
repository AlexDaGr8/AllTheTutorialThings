<template>
  <loading-indicator :loading="loading" class="text-editor">
    <div v-if="!longForm" class="ui input" :class="[{ 'left icon' : !editing },{action: editing}]">
      <input
        v-if="!longForm"
        type="text"
        placeholder="Enter Value"
        @focus="onFocus()"
        @blur="onBlur($event)"
        v-model="text"
      />
 
      <template v-if="editing">
        <button class="ui icon primary button" :disabled="!editing" @click="save()">
          <i class="icon checkmark"></i>
        </button>
        <button class="ui icon button" :disabled="!editing" @click="cancelEdit()">
          <i class="icon remove"></i>
        </button>
      </template>
      <i v-else class="icon pencil"></i>
    </div>
 
    <div v-else class="ui form">
      <textarea
        placeholder="Enter Value"
        @focus="onFocus()"
        @blur="onBlur($event)"
        v-model="text"
        :rows="10"
      ></textarea>
 
      <div v-if="editing" class="ui small bottom attached buttons">
        <button class="ui icon primary button" :disabled="!editing" @click="save()">
          <i class="icon checkmark"></i>
        </button>
        <button class="ui icon button" :disabled="!editing" @click="cancelEdit()">
          <i class="icon remove"></i>
        </button>
      </div>
    </div>
  </loading-indicator>
</template>
 
<script lang="ts">
import Vue from 'vue';
import Util from '@/util/Util';
import LoadingIndicator from '@/components/util/LoadingIndicator.vue';
 
export default Vue.extend({
  components: {
    LoadingIndicator
  },
 
  props: ['value', 'onSave', 'longForm'],
 
  watch: {
    value(newVal, oldVal) {
      this.text = newVal;
    }
  },
 
  data() {
    return {
      loading: false,
      originalValue: this.value,
      text: this.value,
      editing: false
    };
  },
 
  methods: {
    onFocus(): void {
      this.edit();
    },
    onBlur(evt): void {
      if (this.originalValue !== this.text) {
        setTimeout(() => {
          if (!this.loading && this.editing) {
            evt.preventDefault();
            evt.stopPropagation();
 
            this.save();
          }
        }, 200);
      } else {
        this.editing = false;
      }
    },
    edit(): void {
      this.editing = true;
    },
    cancelEdit(keeper = false): void {
      if (!keeper) {
        this.text = this.originalValue;
      }
 
      this.editing = false;
    },
    async save(): Promise<any> {
      let task = Promise.resolve();
      let keeper = false;
 
      const isString = typeof this.text === 'string';
 
E      if (isString && this.onSave) {
        this.loading = true;
 
        task = this.onSave(this.text);
        keeper = true;
      }
 
      return task.then(() => {
        this.cancelEdit(keeper);
        this.loading = false;
      });
    }
  }
});
</script>
