<template>
  <loading-indicator :loading="loading" class="number-editor">
    <div class="ui input" :class="[{ 'left icon' : !editing },{action: editing}]">
      <input
        type="number"
        placeholder="Enter Value"
        @focus="onFocus()"
        @blur="onBlur($event)"
        v-model.number="num"
      />
      <template v-if="editing">
        <button class="ui primary icon button" :disabled="!editing" @click="save()">
          <i class="icon checkmark"></i>
        </button>
        <button class="ui icon button" :disabled="!editing" @click="cancelEdit()">
          <i class="icon remove"></i>
        </button>
      </template>
 
      <i v-else class="icon pencil"></i>
    </div>
  </loading-indicator>
</template>
 
<script lang="ts">
import Vue from 'vue';
import LoadingIndicator from '@/components/util/LoadingIndicator.vue';
 
export default Vue.extend({
  components: {
    LoadingIndicator
  },
 
  props: ['value', 'onSave'],
 
  watch: {
    value(newVal, oldVal) {
      this.num = newVal;
    }
  },
 
  data() {
    return {
      loading: false,
      originalValue: this.value,
      num: this.value,
      editing: false
    };
  },
 
  methods: {
    onFocus(): void {
      this.edit();
    },
    onBlur(evt): void {
      if (this.originalValue !== this.num) {
        setTimeout(() => {
          if (!this.loading && this.editing) {
            evt.preventDefault();
            evt.stopPropagation();
 
            this.save();
          }
        }, 200);
      } else {
        this.editing = false;
      }
    },
    edit(): void {
      this.editing = true;
    },
    cancelEdit(keeper = false): void {
      if (!keeper) {
        this.num = this.originalValue;
      }
 
      this.editing = false;
    },
    async save(): Promise<any> {
      let task = Promise.resolve();
      let keeper = false;
 
      if (this.onSave) {
        this.loading = true;
        // @ts-ignore
        task = this.onSave(this.num);
        keeper = true;
      }
 
      return task.then(() => {
        this.cancelEdit(keeper);
        this.loading = false;
      });
    }
  }
});
</script>
