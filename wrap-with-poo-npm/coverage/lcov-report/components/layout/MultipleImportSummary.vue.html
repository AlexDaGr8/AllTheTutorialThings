<template>
  <two-columns-layout left-column-width="eight" :space-between="false" class="divided">
    <template v-slot:left-column>
      <p class="big-font">
        {{ successList.length }} {{resultsMessage(successList)}}
        <strong
          class="success-color"
        >successfully</strong>
        {{itemTask}}.
      </p>
 
      <ol class="word-break">
        <li v-for="(item, index) in successList" :key="`${item.value}-${index}`">
          <strong>{{item.value}}</strong>
        </li>
      </ol>
    </template>
    <template v-slot:right-column v-if="failList && failList.length">
      <p class="big-font">
        {{ failList.length }} {{ resultsMessage(failList)}}
        <strong class="danger-color">not</strong>
        {{itemTask}}.
      </p>
      <ol class="word-break">
        <li v-for="(item, index) in failList" :key="`${item.value}-${index}`">
          <strong>{{item.value}}</strong>
          - {{item.error}}
        </li>
      </ol>
    </template>
  </two-columns-layout>
</template>
<script lang="ts">
import Vue from 'vue';
import Core from '@/types/core.d.ts';
 
export default Vue.extend({
  name: 'MultipleImportSummary',
  props: {
    itemName: {
      type: String,
      default: 'item'
    },
    itemTask: {
      type: String,
      default: 'saved'
    },
    successList: {
      type: Array as () => Core.CsvItem[],
      required: false
    },
    failList: {
      type: Array as () => Core.CsvItem[],
      required: false
    }
  },
  methods: {
    resultsMessage(list) {
      return list.length > 1
        ? `${this.itemName}s were`
        : `${this.itemName} was`;
    }
  }
});
</script>
