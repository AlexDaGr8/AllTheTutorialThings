<template>
  <span>
    <i v-if="iconPopup" ref="iconPopover" class="simple-popover" :class="icon + ' icon link'"></i>
    <a v-else-if="!iconPopup" ref="popover" class="simple-popover">
      <slot name="popover"></slot>
    </a>
  </span>
</template>
 
<script lang="ts">
import Vue from 'vue';
declare var $: any;
 
export default Vue.extend({
  name: 'SimplePopover',
  props: {
    popupClasses: {
      type: String,
      required: false,
      default: ''
    },
    content: {
      type: String,
      required: true
    },
    title: {
      type: String
    },
    iconPopup: {
      type: Boolean,
      default: true
    },
    icon: {
      type: String,
      default: 'info circle'
    },
    triggerOn: {
      type: String,
      default: 'hover',
      validator: (value: string) => {
        return ['click', 'hover'].indexOf(value) !== -1;
      }
    },
    position: {
      type: String,
      default: 'right center',
      validator: (value: string) => {
        return (
          [
            'top left',
            'top center',
            'top right',
            'bottom left',
            'bottom center',
            'bottom right',
            'right center',
            'left center'
          ].indexOf(value) !== -1
        );
      }
    },
    width: {
      type: String,
      validator: (value: string) => {
        return ['wide', 'very wide'].indexOf(value) !== -1;
      }
    },
    size: {
      type: String,
      validator: (value: string) => {
        return ['mini', 'tiny', 'small', 'large', 'huge'].indexOf(value) !== -1;
      }
    },
    inverted: {
      type: Boolean
    },
    variation: {
      type: String,
      validator: (value: string) => {
        return ['basic'].indexOf(value) !== -1;
      }
    }
  },
  mounted() {
    this.$nextTick(() => {
      const popoverConfig = {
        className: {
          popup: `ui popup ${this.popupClasses}`
        },
        title: this.$props.title,
        content: this.$props.content,
        on: this.$props.triggerOn,
        position: this.$props.position,
        lastResort: true,
        variation: `${this.$props.variation} ${this.$props.width} ${
          this.$props.size
        } ${this.$props.inverted ? 'inverted' : ''}`
      };
      $(this.$refs.iconPopover).popup(popoverConfig);
      $(this.$refs.popover).popup(popoverConfig);
    });
  }
});
</script>
