<template>
  <header id="page-header" class="ui breadcrumb sixteen wide column padding-bottom-2 padding-top-4">
    <h1 class="ui header">
      <i
        v-if="pageImage"
        class="icon lg"
        :class="pageHeader.icon ? pageHeader.icon : ''"
        :style="pageHeader.svg ? {  '-webkit-mask': `url(${pageHeader.svg}) no-repeat center`, 'mask': `url(${pageHeader.svg}) no-repeat center`, backgroundColor: 'var(--heading-color)'} : null"
      >
        <img v-if="pageHeader.img" class="ui image" :src="pageHeader.img" />
      </i>
      <span class="content">
        <span class="truncated-page-header">
          <template v-if="pageHeader.parent_path">
            <router-link :to="pageHeader.parent_path">{{ pageHeader.parent_title }}</router-link>
            <i class="right chevron icon divider"></i>
          </template>
          {{beforeTitle}} {{ pageHeader.title }}
        </span>
        <div class="sub header">{{ pageHeader.description }}</div>
      </span>
    </h1>
  </header>
</template>
 
<script lang="ts">
import Vue from 'vue';
import RouteHelper from '@/util/RouteHelper.ts';
import Core from '@/types/core.d.ts';
 
export default Vue.extend({
  name: 'PageHeader',
  props: {
    name: String,
    beforeTitle: String,
    includePageTitle: { type: Boolean, default: true }
  },
  computed: {
    pageName(): string {
      return this.name ? ' ' + this.name : '';
    },
    pageImage(): boolean {
      return (
        this.pageHeader.icon || this.pageHeader.image || this.pageHeader.svg
      );
    },
    pageHeader(): Core.PageHeader {
      return RouteHelper.getPageHeaderInfo(
        this.$route,
        this.pageName,
        this.includePageTitle
      );
    }
  }
});
</script>
