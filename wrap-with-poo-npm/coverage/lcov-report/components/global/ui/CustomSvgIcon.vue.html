  <i class="ui icon custom-svg-icon" :class="sizeClass" :style="styles"></i>

<script lang="ts">
import Vue from 'vue';
export default Vue.extend({
  name: 'CustomSvgIcon',
  props: {
    sizeClass: {
      type: String,
      default: null
    },
    sizeEm: {
      type: Number,
      default: null
    },
    svgIcon: {
      type: String,
      default: ''
    },
    color: {
      type: String,
      default: '--brand'
    },
    colorHover: {
      type: String,
      default: '--brand-dark'
    }
  },
  methods: {
    colorStyle(): string {
      return this.color.indexOf('--') > -1 ? `var(${this.color})` : this.color;
    },
    colorHoverStyle(): string {
      return this.colorHover.indexOf('--') > -1
        ? `var(${this.colorHover})`
        : this.colorHover;
    }
  },
  computed: {
    fontSize(): string {
      return `${this.sizeEm}em`;
    },
    styles() {
      let styleList = {};
      styleList = Object.assign(styleList, {
        '-webkit-mask': `url(${this.svgIcon}) no-repeat center`,
        mask: `url(${this.svgIcon}) no-repeat center`,
        '--background-color': `${this.colorStyle()}`,
        '--background-color-hover': `${this.colorHoverStyle()}`
      });
E      if (this.sizeEm) {
        styleList = Object.assign(styleList, {
          'font-size': this.fontSize
        });
      }
      return styleList;
    }
  }
});
</script>
