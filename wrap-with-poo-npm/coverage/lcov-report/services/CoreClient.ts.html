  async getActionLog(
    dateRange: Core.DateRange | null,
    textFilter: string | null,
    pageRequest: Core.PageRequest | null
  ): Promise<Core.PagedItems> {
    let query = `?`;
 
    if (dateRange) {
      if (dateRange.start) {
        query += `&startDate=${dateRange.start}`;
      }
      if (dateRange.end) {
        query += `&endDate=${dateRange.end}`;
      }
    }
 
    if (textFilter) {
      query += `&textFilter=${textFilter}`;
    }
 
    if (pageRequest) {
      query += `&pageNumber=${pageRequest.pageNumber}`;
      query += `&pageSize=${pageRequest.pageSize}`;
