import Client from '@/services/Client';
import Core from '@/types/core.d.ts';
 
const root = '/admin-service/';
 
class SettingClient extends Client {
  async getSettings(): Promise<any> {
    const resp = await this.http.get(``);
    return resp.data.data;
  }
 
  async updateSetting(setting: Core.Setting): Promise<any> {
    const uri = `application/${setting.application}/${setting.name}/scope/customer`;
 
    try {
      const resp = await this.http.put(uri, {
        value: setting.value
      });
 
      if (resp.data.isAxiosError) {
        throw new Error(resp.data.message);
      }
 
      return resp.data;
    } catch (ex) {
      throw ex;
    }
  }
 
  async deleteSetting(setting: Core.Setting): Promise<any> {
    const uri = `application/${setting.application}/${setting.name}/scope/customer`;
 
    try {
      const resp = await this.http.delete(uri);
 
      if (resp.data.isAxiosError) {
        throw new Error(resp.data.message);
      }
 
      return resp.data;
    } catch (ex) {
      throw ex;
    }
  }
}
 
class AdminClient {
  settings: SettingClient;
 
  constructor() {
    this.settings = new SettingClient(root, 'application-setting');
  }
}
 
export default new AdminClient();
