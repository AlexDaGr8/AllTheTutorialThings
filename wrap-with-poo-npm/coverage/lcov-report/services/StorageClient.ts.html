import Client from '@/services/Client';
import Core from '@/types/core.d.ts';
 
class QuarantineClient extends Client {
  constructor(root: string = '/api/storage/v1/', basePath: string = 'quarantine') {
  super(root, basePath);
}
 
  async getQuarantineItems(
    dateRange: Core.DateRange | null,
    status: string | null,
    text: string | null,
    pageRequest: Core.PageRequest
  ): Promise<Core.PagedItems> {
    let query = `?`;
 
    if (dateRange) {
      if (dateRange.start) {
        query += `&startDate=${dateRange.start}`;
      }
      if (dateRange.end) {
        query += `&endDate=${dateRange.end}`;
      }
    }
 
    if (text) {
      query += `&text=${text}`;
    }
 
    if (status) {
      query += `&status=${status}`;
    }
 
    if (pageRequest) {
      query += `&pageNumber=${pageRequest.pageNumber}`;
      query += `&pageSize=${pageRequest.pageSize}`;
    }
 
    const resp = await this.http.get(query);
    return resp.data;
  }
 
  async getQuarantineItemByID(quarantineId: number): Promise<Core.QuarantineItem> {
    const resp = await this.http.get(`${quarantineId}`);
    return resp.data;
  }
 
  async getQuarantineHistory(quarantineId: number): Promise<Core.QuarantineItemHistory> {
    const resp = await this.http.get(`${quarantineId}/history`);
    return resp.data.data;
  }
 
  getDownloadUri(quarantineId: number) {
    return `${this.baseApi}${quarantineId}/download` ;
  }
 
  /**
   * Release a quarantine with the given id
   */
  async release(quarantineId: number) {
    const resp = await this.http.post(`${quarantineId}/release`, null);
    return resp.data;
  }
 
  /**
   * Delete a quarantine with the given id.  This action will set quarantine to delete status and hard delete the file.
   */
  async delete(quarantineId: number) {
    const resp = await this.http.post(`${quarantineId}/delete`, null);
    return resp.data;
  }
 
}
 
class StorageClient {
  quarantine: QuarantineClient;
 
  constructor() {
    this.quarantine = new QuarantineClient();
  }
}
 
export default new StorageClient();
