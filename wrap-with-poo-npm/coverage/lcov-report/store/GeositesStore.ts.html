  upsertGeosite(
    state: GeositesState,
    geosite: Core.Geosite | Core.GeositeLocation
  ): void {
    const existing = state.geosites.find(g => g.id === geosite.id);
    if (!existing) {
      state.geosites.push(<Core.Geosite>geosite);
    } else {
      const updatedGeosite: Core.Geosite = Object.assign(existing, geosite);

  updateLocationInGeosites(
    state: GeositesState,
    geositeLocation: Core.GeositeLocation
  ): void {

    const allLocations = state.geosites.filter(
      (geosite: Core.Geosite) =>
