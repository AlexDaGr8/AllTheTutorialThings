import Vue from 'vue';
import CoreClient from '@/services/CoreClient';
import Core from '@/types/core.d.ts';
 
export interface TenantsState {
  authDomain: string;
  tenants: string[];
}
 
export enum TenantMutations {
  SET_TENANTS = 'setTenants',
  SET_AUTH_DOMAIN = 'setAuthDomain'
}
 
export enum TenantActions {
  GET_TENANTS = 'getTenants'
}
 
export const mutations = {
  setTenants(state: TenantsState, tenants: string[]): void {
    Vue.set(state, 'tenants', tenants);
  },
  setAuthDomain(state: TenantsState, authDomain: string): void {
    Vue.set(state, 'authDomain', authDomain);
  }
};
 
export const actions = {
  async getTenants({ commit }) {
    const resp = await CoreClient.tenants.getTenants();
 
    commit(TenantMutations.SET_TENANTS, resp.tenants);
    commit(TenantMutations.SET_AUTH_DOMAIN, resp.authDomain);
  }
};
 
export const getters = {};
 
const TenantsStore = {
  state: {
    tenants: []
  },
  mutations,
  actions,
  getters
};
 
export default TenantsStore;
