import Vue from 'vue';
import CoreClient from '@/services/CoreClient';
import Core from '@/types/core.d.ts';
 
export interface ActionLogState {
  logItems: Core.ActionLogItem[];
  pagerState: Core.PagerState;
}
 
export enum ActionLogMutations {
  SET_LOG_ITEMS = 'setLogItems',
  SET_LOG_PAGE_NUMBER = 'setLogPageNumber',
  SET_LOG_PAGE_SIZE = 'setLogPageSize',
  SET_LOG_TOTAL_SIZE = 'setLogTotalSize',
  CLEAR_LOG_ITEMS = 'clearLogItems'
}
 
export enum ActionLogActions {
  GET_LOG_ITEMS = 'getLogItems'
}
 
export enum ActionLogGetters {
  LOG_ITEMS = 'logItems',
  PAGER_STATE = 'pagerState'
}
 
export const mutations = {
  setLogItems(state: ActionLogState, items: Core.ActionLogItem[]): void {
    Vue.set(state, 'logItems', items);
  },
  setLogPageNumber(state: ActionLogState, pageNumber: number): void {
    Vue.set(state.pagerState, 'pageNumber', pageNumber);
  },
  setLogPageSize(state: ActionLogState, pageSize: number): void {
    Vue.set(state.pagerState, 'pageSize', pageSize);
  },
  setLogTotalSize(state: ActionLogState, totalSize: number): void {
    Vue.set(state.pagerState, 'totalSize', totalSize);
  },
  clearLogItems(state: ActionLogState): void {
    Vue.set(state, 'logItems', []);
  }
};
 
export const actions = {
  async getLogItems({ commit }, { dateRange, textFilter, pageRequest }) {
    const resp: Core.PagedItems = await CoreClient.actionLog.getActionLog(
      dateRange,
      textFilter,
      pageRequest
    );
 
    commit(ActionLogMutations.SET_LOG_ITEMS, resp.data);
    commit(ActionLogMutations.SET_LOG_PAGE_NUMBER, resp.pageNumber);
    commit(ActionLogMutations.SET_LOG_PAGE_SIZE, resp.pageSize);
    commit(ActionLogMutations.SET_LOG_TOTAL_SIZE, resp.totalSize);
  }
};
 
export const getters = {
  logItems: (state: ActionLogState) => {
    return state.logItems;
  },
  pagerState: (state: ActionLogState) => {
    return state.pagerState;
  }
};
 
const ActionItemStore = {
  state: {
    logItems: [],
    pagerState: {
      pageNumber: 0,
      pageSize: 15,
      totalSize: 0
    }
  },
  mutations,
  actions,
  getters
};
 
export default ActionItemStore;
