import Vue from 'vue';
import CoreClient from '@/services/CoreClient';
import Core from '@/types/core.d.ts';
 
export interface LocaleState {
  countryCodes: Core.CountryCode[];
  timezones: Core.Timezone[];
}
 
export enum LocaleMutations {
  SET_COUNTRY_CODES = 'setCountryCodes',
  CLEAR_COUNTRY_CODES = 'clearCountryCodes',
 
  SET_TIMEZONES = 'setTimezones',
  CLEAR_TIMEZONES = 'clearTimezones'
}
 
export enum LocaleActions {
  GET_COUNTRY_CODES = 'getCountryCodes',
  GET_TIMEZONES = 'getTimezones'
}
 
export const mutations = {
  setCountryCodes(state: LocaleState, countryCodes: Core.CountryCode[]): void {
    Vue.set(state, 'countryCodes', countryCodes);
  },
  clearCountryCodes(state: LocaleState): void {
    Vue.set(state, 'countryCodes', []);
  },
  setTimezones(state: LocaleState, timezones: Core.Timezone[]): void {
    Vue.set(state, 'timezones', timezones);
  },
  clearTimezones(state: LocaleState): void {
    Vue.set(state, 'timezones', []);
  }
};
 
export const actions = {
  async getCountryCodes({ commit }) {
    const resp: Core.CountryCode[] = await CoreClient.locale.getCountryCodes();
 
    commit(LocaleMutations.SET_COUNTRY_CODES, resp);
  },
  async getTimezones({ commit }) {
    const resp: Core.Timezone[] = await CoreClient.locale.getTimezones();
 
    commit(LocaleMutations.SET_TIMEZONES, resp);
  }
};
 
export const getters = {
  getTimezoneById: (state: LocaleState) => (timezoneId: number) => {
    return state.timezones.find(tz => tz.id === timezoneId);
  }
};
 
const LocaleStore = {
  state: {
    countryCodes: [],
    timezones: []
  },
  mutations,
  actions,
  getters
};
 
export default LocaleStore;
