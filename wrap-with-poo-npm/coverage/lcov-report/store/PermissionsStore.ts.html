export const mutations = {
  setPermissions(
    state: PermissionsState,
    permissions: Core.Permission[]
  ): void {
    permissions.forEach(p => {
      mutations.upsertPermission(state, p);
    });
  },
  upsertPermission(state: PermissionsState, permission: Core.Permission): void {
    const existing = state.permissions.find(p => p.id === permission.id);
    if (!existing) {
      state.permissions.push(permission);
    } else {
      const updatedPermission = Object.assign(existing, permission);
      state.permissions.splice(
        state.permissions.indexOf(existing),
        1,
        updatedPermission
      );
    }
  },
  removePermission(state: PermissionsState, permissionId: number): void {
    Vue.set(
      state,
      'permissions',
      state.permissions.filter(p => {
        return p.id !== permissionId;
      })
    );
  },
  clearPermissions(state: PermissionsState): void {
    Vue.set(state, 'permissions', []);
  }
};
