export enum PERSONA_NOTIFICATION_EVENT {
  CREATED = 'created',
  ACTIVATED = 'activated',
  DEACTIVATED = 'deactivated',
  UPDATED = 'updated',
  DELETED = 'deleted'
}
 
export type PersonaNotificationEvent =

  | PERSONA_NOTIFICATION_EVENT.DELETED;
 
export const mutations = {
  setFetchError(state: PersonasState, error: string | null): void {
    Vue.set(state, 'fetchError', error);
  },
  setPersonas(state: PersonasState, personas: Core.Persona[]): void {
    Vue.set(state, 'personas', personas);
  },
  upsertPersona(state: PersonasState, persona: Core.Persona): void {
    const existing = state.personas.find(p => p.id === persona.id);
    if (existing) {
      const updatedPersona = Object.assign(existing, persona);
      state.personas.splice(
        state.personas.indexOf(existing),
        1,
        updatedPersona
      );
    } else {
      state.personas.push(persona);
    }
  },
  setCsvPreviewResult(state: PersonasState, csvPreviewResult: any): void {
    // reorder the details to match the schema
    csvPreviewResult.map((item: any) => {
      const { details } = item;
      const reorderedDetails = {};
      // codename is a part of the schema but not a detail
      // we do not want to include it in the reorder
      const [codename, ...schema] = state.schema;
      schema.forEach(key => reorderedDetails[key] = details[key]);
      item.details = { ...reorderedDetails, ...details };
      return item;
    });
 
    Vue.set(
      state,
      'csvPreviewResult',
      PotentialPersonasExtractor.extract(csvPreviewResult)
    );
  },
  clearCsvPreviewResult(state: PersonasState): void {
    Vue.set(state, 'csvPreviewResult', null);
  },
  setCsvResult(state: PersonasState, csvResult: any): void {
    Vue.set(state, 'csvResult', PotentialPersonasExtractor.extract(csvResult));
  },
  clearCsvResult(state: PersonasState): void {
    Vue.set(state, 'csvResult', null);
  },
  removePersona(state: PersonasState, personaId: number): void {
    Vue.set(
      state,
      'personas',
      state.personas.filter(p => {
        return p.id !== personaId;
      })
    );
  },
  clearPersonas(state: PersonasState): void {
    Vue.set(state, 'personas', []);
  },
  setSchema(state: PersonasState, schema: string[]): void {
    Vue.set(state, 'schema', schema);
  }
};

export const getters = {
  getPersonaById: (state: PersonasState) => (personaId: number) => {
    const persona = state.personas.find(
      (p: Core.Persona) => p.id === personaId
    );
    return persona;
  },
  getUserPersonas: (state: PersonasState, getters) => (userId: number) => {
    const userPermissions: Core.Permission[] = getters.getUserPermissions(
      userId
    );
 
    const assignedPersonas: Core.Persona[] = [];
    const availablePersonas: Core.Persona[] = [];
 
    state.personas.forEach(persona => {
      let assigned = false;
 
      userPermissions.forEach(perm => {
        const isPersonaPerm =
          perm.permissionType.namespace === PERMISSION_NAMESPACE.WORKSPACE;
        const canUse = perm.permissionType.name === PERMISSION_NAME.USE;
        const matchingEntity = perm.entityId === persona.id;
 
        if (isPersonaPerm && canUse && matchingEntity) {
          assignedPersonas.push(persona);
          assigned = true;
        }
      });
 
      if (!assigned) {
        availablePersonas.push(persona);
      }
    });
 
    return {
      assigned: assignedPersonas,
      available: availablePersonas
    };
  }
};
