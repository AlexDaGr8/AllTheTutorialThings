/*
  Vue router does not place dynamically added
  routes to the $route.options.routes array so
  this store is necessary so that we can track
  those
 
  https://github.com/vuejs/vue-router/issues/1859
*/
 
import Vue from 'vue';
 
export interface RoutesState {
  routes: any[];
}
 
export enum RouteMutations {
  UPSERT_ROUTES = 'upsertRoutes',
  CLEAR_ROUTES = 'clearRoutes'
}
 
export const mutations = {
  upsertRoutes(state: RoutesState, routes: any[]): void {
    routes.forEach(route => {
      const existing = state.routes.find(r => r.name === route.name);
      if (!existing) {
        state.routes.push(route);
      } else {
        const updatedRoute = Object.assign(existing, route);
        state.routes.splice(state.routes.indexOf(existing), 1, updatedRoute);
      }
    });
  },
  clearRoutes(state: RoutesState): void {
    Vue.set(state, 'routes', []);
  }
};
 
export const actions = {};
 
export const getters = {};
 
const RoutesStore = {
  state: {
    routes: []
  },
  mutations,
  actions,
  getters
};
 
export default RoutesStore;
